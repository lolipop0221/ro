Stack Code Example
stack menggunakan array 
#include <iostream>
using namespace std;

int s[5], t=-1;

void push() {
    if(t==4) cout<<"Stack penuh!\n";
    else { cin>>s[++t]; cout<<"Ditambahkan\n"; }
}

void pop() {
    if(t==-1) cout<<"Stack kosong!\n";
    else cout<<s[t--]<<" dihapus\n";
}

void lihat() {
    if(t==-1) cout<<"Stack kosong\n";
    else {
        cout<<"Stack: ";
        for(int i=t;i>=0;i--) cout<<s[i]<<" ";
        cout<<endl;
    }
}

int main() {
    int p;
    while(1) {
        cout<<"\n1.Push\n2.Pop\n3.Lihat\n4.Keluar\nPilih: ";
        cin>>p;
        if(p==4) break;
        switch(p) {
            case 1: push(); break;
            case 2: pop(); break;
            case 3: lihat(); break;
            default: cout<<"Salah!\n";
        }
    }
    return 0;
}
stack menggunakan linked list 
#include <iostream>
using namespace std;

struct Node {
    int nilai;
    Node* berikutnya;
};

Node* atas = nullptr;

void push() {
    Node* nodeBaru = new Node();
    cout << "Masukkan nilai: ";
    cin >> nodeBaru->nilai;
    nodeBaru->berikutnya = atas;
    atas = nodeBaru;
    cout << "Ditambahkan\n";
}

void pop() {
    if (atas == nullptr) {
        cout << "Stack kosong!\n";
    } else {
        Node* temp = atas;
        cout << temp->nilai << " dihapus\n";
        atas = atas->berikutnya;
        delete temp;
    }
}

void lihat() {
    if (atas == nullptr) {
        cout << "Stack kosong\n";
    } else {
        cout << "Stack: ";
        Node* saatIni = atas;
        while (saatIni != nullptr) {
            cout << saatIni->nilai << " ";
            saatIni = saatIni->berikutnya;
        }
        cout << endl;
    }
}

int main() {
    int pilihan;
    while(true) {
        cout << "\n1.Push\n2.Pop\n3.Lihat\n4.Keluar\nPilih: ";
        cin >> pilihan;
        if(pilihan == 4) break;
        switch(pilihan) {
            case 1: push(); break;
            case 2: pop(); break;
            case 3: lihat(); break;
            default: cout << "Pilihan salah!\n";
        }
    }
    
    while(atas != nullptr) {
        Node* temp = atas;
        atas = atas->berikutnya;
        delete temp;
    }
    
    return 0;
}
stack menggunakan STL 
#include <iostream>
#include <stack>
using namespace std;

stack<int> st;

void push() {
    int value;
    cout << "Masukkan angka: ";
    cin >> value;
    st.push(value);
}

void pop() {
    if (st.empty()) {
        cout << "Stack kosong!\n";
        return;
    }
    cout << st.top() << " dihapus\n";
    st.pop();
}

void show() {
    if (st.empty()) {
        cout << "Stack kosong\n";
        return;
    }

    stack<int> temp = st;
    cout << "Isi stack: ";
    
    while (!temp.empty()) {
        cout << temp.top() << " ";
        temp.pop();
    }
    cout << endl;
}

int main() {
    int choice;
    while (true) {
        cout << "\n1. Push\n2. Pop\n3. Lihat\n4. Keluar\nPilihan: ";
        cin >> choice;
        
        if (choice == 4) break;
        
        switch (choice) {
            case 1: push(); break;
            case 2: pop(); break;
            case 3: show(); break;
            default: cout << "Input salah!\n";
        }
    }
    return 0;
}
